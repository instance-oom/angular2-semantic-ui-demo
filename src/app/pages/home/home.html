Get Started
A angular2 component set of semantic-ui, and no jQuery dependency

Install

    npm install angular2-semantic-ui --save

Usage

import << L_SEMANTIC_UI_MODULE >> from 'angular2-semantic-ui';

@NgModule(<<
  declarations: [
    APP
  ],
  imports: [
    BrowserModule,
    CommonModule,
    FormsModule,
    ReactiveFormsModule,
    L_SEMANTIC_UI_MODULE
  ],
  providers: [],
  bootstrap: [APP]
>>)
export class AppModule << >>
